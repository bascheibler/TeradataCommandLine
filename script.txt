.RUN FILE = logon.txt

SELECT 1 FROM DBC.TABLES WHERE DatabaseName = 'BI_INDICADORES' AND TableName = 'TESTE';

.IF ACTIVITYCOUNT = 0 THEN GOTO CreateNewTable

DROP TABLE BI_INDICADORES.TESTE;

.IF ERRORCODE = 3807 THEN GOTO CreateNewTable
.IF ERRORCODE > 0 THEN .QUIT ERRORCODE

.LABEL CreateNewTable

CREATE TABLE BI_INDICADORES.TESTE AS (SELECT 'A' AS A) WITH DATA;

.LOGOFF
.EXIT
